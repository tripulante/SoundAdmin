00412 // This section contains the data structure that is used to determine how
00413 // often to print a report for each sensor of each tracker.  Each element
00414 // contains the number of reports to skip between printing and a counter
00415 // that is used by the callback routine to keep track of how many it has
00416 // skipped.  There is an element for each possible sensor.  A new array of
00417 // elements is created for each new tracker object, and a pointer to it is
00418 // passed as the userdata pointer to the callback handlers.  A separate
00419 // array is kept for the position, velocity, and acceleration for each
00420 // tracker.  The structure passed to the callback handler also has a
00421 // string that is the name of the tracker.
00422 
00423 class t_sensor_counter {
00424     public:
00425         int stride;     // Stride of 1 == print every, 2 is every other, ...
00426         int count;      // How many we have gotten
00427 };

00429     public:
00430         char                t_name[vrpn_MAX_TEXT_LEN];
